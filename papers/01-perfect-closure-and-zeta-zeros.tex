\begin{abstract}
This entry paper defines Perfect Closure in the zeta setting and isolates the key local theorem: for each prime route, the critical line $\Re(s)=\tfrac12$ is the unique amplitude-balance line. With
\[
B_p(\sigma)=\left|p^{-\sigma}-p^{-(1-\sigma)}\right|,
\]
one gets
\[
B_p\!\left(\frac12+\delta\right)=2p^{-1/2}|\sinh(\delta\log p)|,
\]
so $B_p(\sigma)=0$ iff $\sigma=\tfrac12$. We then set $\Xi(t)=\xi(\tfrac12+it)$ and $C(t)=|\Xi(t)|^2$, using $\Xi(t_n)=0$ as the complex spectral trace condition carried by later papers. No proof of RH is claimed.
\end{abstract}

\section*{Reader Guide}
This paper answers one question: what is the complex spectral trace? The point is not to locate zeros, but to identify $\Xi(t_n)=0$ as the completed critical-line residual condition used later.

\section{Perfect Closure in the zeta setting}
Perfect Closure means vanishing of a selected residual built from zeta data.
Local residual: $B_p$. Global residual: $C(t)=|\Xi(t)|^2$.

\Needspace{10\baselineskip}
\section{Prime route and square-root balance}
For prime $p>1$ and $s=\sigma+it$,
\[
p^{-s}=p^{-\sigma}e^{-it\log p}.
\]
On the critical line,
\[
p^{-(1/2+it)}=\frac1{\sqrt p}e^{-it\log p},\qquad
p^{-(1/2-it)}=\frac1{\sqrt p}e^{+it\log p}.
\]
In plain language, the critical line is the balance line where conjugate prime-route amplitudes match.

\Needspace{10\baselineskip}
\section{Local imbalance theorem}
Define
\[
B_p(\sigma)=\left|p^{-\sigma}-p^{-(1-\sigma)}\right|.
\]
With $\sigma=\tfrac12+\delta$,
\[
B_p\!\left(\frac12+\delta\right)=2p^{-1/2}|\sinh(\delta\log p)|.
\]

\begin{figure}[htbp]
  \centering
  \includegraphics[width=0.80\textwidth]{../assets/figures/prime-imbalance-curve.png}
  \caption{Illustrative imbalance curves for $B_p(1/2+\delta)=2p^{-1/2}|\sinh(\delta\log p)|$ at $p=2,5,11$, with $\delta=0$ shown as the zero-imbalance point. This visual supports Proposition 3.1 qualitatively and is not a numerical claim about zero locations.}
\end{figure}

\begin{proposition}
For every $p>1$, $B_p(\sigma)=0$ iff $\sigma=\tfrac12$.
\end{proposition}

\begin{proof}
Since $p^{-1/2}>0$,
\[
B_p\!\left(\frac12+\delta\right)=0\iff \sinh(\delta\log p)=0.
\]
Because $\log p\neq0$ and $\sinh x=0$ iff $x=0$, we get $\delta=0$, i.e. $\sigma=\tfrac12$. The converse is immediate.
\end{proof}

\Needspace{10\baselineskip}
\section{Optional finite-prime support}
For primes $p_1<\cdots<p_N$, let
\[
B_N(\sigma)^2=\sum_{j=1}^{N}B_{p_j}(\sigma)^2.
\]
Then $B_N(\sigma)=0$ iff $\sigma=\tfrac12$ by applying Proposition 3.1 to each nonnegative term.

\begin{remark}
This finite-prime statement is supporting context, not the main endpoint of the paper.
\end{remark}

In $\Re(s)>1$,
\[
\log\zeta(s)=\sum_p\sum_{k\ge1}\frac{p^{-ks}}{k}.
\]
We mention this only as bookkeeping context.

\section{Completed residual and complex spectral trace}
Define
\[
\Xi(t)=\xi\!\left(\frac12+it\right),\qquad C(t)=|\Xi(t)|^2.
\]
The complex closure condition used later is
\[
\Xi(t_n)=0.
\]
