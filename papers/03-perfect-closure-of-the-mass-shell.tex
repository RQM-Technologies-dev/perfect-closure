\begin{abstract}
This paper defines the mass shell in quaternionic norm language. With biquaternionic momentum
\[
P=\frac{E}{c}+I\mathbf p,\qquad I^2=-1,
\]
the invariant relation becomes
\[
P\bar P=m^2c^2.
\]
Null modes satisfy $P\bar P=0$ and massive modes satisfy $P\bar P>0$. We also record Compton curvature $k_C=mc/\hbar$. The zeta bridge is deferred to Paper 4.
\end{abstract}

\section{Relativistic mass shell}
\[
E^2=p^2c^2+m^2c^4,
\qquad
\frac{E^2}{c^2}-|\mathbf p|^2=m^2c^2.
\]

\Needspace{10\baselineskip}
\section{Biquaternionic momentum norm}
Define
\[
P=\frac{E}{c}+I\mathbf p.
\]
Then
\[
P\bar P=\left(\frac{E}{c}\right)^2-|\mathbf p|^2,
\qquad
P\bar P=m^2c^2.
\]

\Needspace{10\baselineskip}
\section{Open/null and closed/massive interpretation}
\begin{remark}
$P\bar P=0$ corresponds to null/open propagation, while $P\bar P>0$ corresponds to massive/closed invariant norm. This is kinematic, not yet predictive dynamics.
\end{remark}
