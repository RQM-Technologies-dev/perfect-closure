\begin{abstract}
We formalize a closure interpretation of the relativistic mass shell in biquaternionic variables. Writing $P=E/c+I\mathbf p$ with commuting scalar imaginary $I^2=-1$, the invariant relation becomes $P\bar P=m^2c^2$. Null modes satisfy $P\bar P=0$ and massive modes satisfy $P\bar P>0$. This is an interpretation and formalization, not a derivation of Standard Model masses.
\end{abstract}

\section{Introduction}

\section{Standard relativistic mass shell}
\[
E^2=p^2c^2+m^2c^4,
\qquad
\frac{E^2}{c^2}-|\mathbf p|^2=m^2c^2.
\]
$m=0$ is null/lightlike; $m>0$ gives nonzero invariant norm.

\section{Biquaternionic spacetime}

Let $\mathbf r=xi+yj+zk$ and $I^2=-1$ commute with $i,j,k$.
Define
\[
X=ct+I\mathbf r.
\]
Then
\[
X\bar X=(ct+I\mathbf r)(ct-I\mathbf r)=c^2t^2-|\mathbf r|^2.
\]

\begin{remark}
Biquaternionic/paravector structure is used to encode Minkowski signature.
\end{remark}

\section{Biquaternionic momentum}

Define
\[
P=\frac{E}{c}+I\mathbf p,
\]
with $\mathbf p=p_x i+p_y j+p_z k$. Then
\[
P\bar P=\frac{E^2}{c^2}-|\mathbf p|^2,
\]
so the mass shell is
\[
P\bar P=m^2c^2.
\]

\section{Closure interpretation}

\begin{proposition}
$P\bar P=0$ corresponds to null/open flow; $P\bar P>0$ corresponds to massive/closed invariant flow.
\end{proposition}

\begin{proof}
Directly from $P\bar P=E^2/c^2-|\mathbf p|^2$, null propagation has zero Minkowski norm, while massive states have positive invariant norm $m^2c^2$.
\end{proof}
