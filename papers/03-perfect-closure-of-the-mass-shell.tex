\begin{abstract}
We formalize a closure interpretation of the relativistic mass shell in biquaternionic variables. Writing $P=E/c+I\mathbf p$ with commuting scalar imaginary $I^2=-1$, the invariant relation becomes $P\bar P=m^2c^2$. Null modes satisfy $P\bar P=0$ and massive modes satisfy $P\bar P>0$. This is a formal interpretation, not a derivation of known particle masses.
\end{abstract}

\section{Standard relativistic mass shell}
\[
E^2=p^2c^2+m^2c^4,
\qquad
\frac{E^2}{c^2}-|\mathbf p|^2=m^2c^2.
\]
These equations anchor notation in standard physics; they do not introduce a zeta claim.

\section{Biquaternionic spacetime}
Let $\mathbf r=xi+yj+zk$ and $I^2=-1$ commute with $i,j,k$. Define
\[
X=ct+I\mathbf r.
\]
Then
\[
X\bar X=(ct+I\mathbf r)(ct-I\mathbf r)=c^2t^2-|\mathbf r|^2.
\]

\begin{remark}
This is a compact algebraic representation of Minkowski signature, not a new dynamics.
\end{remark}

\section{Biquaternionic momentum and norm}
Define
\[
P=\frac{E}{c}+I\mathbf p.
\]
Then
\[
P\bar P=\left(\frac{E}{c}\right)^2-|\mathbf p|^2,
\qquad
P\bar P=m^2c^2.
\]
This is the physical norm form used by Paper 4.

\section{Closure interpretation}
\begin{proposition}
$P\bar P=0$ corresponds to null/open flow; $P\bar P>0$ corresponds to massive/closed invariant flow.
\end{proposition}

\begin{proof}
Directly from $P\bar P=E^2/c^2-|\mathbf p|^2$, null propagation has zero invariant norm while massive propagation has positive norm $m^2c^2$.
\end{proof}
