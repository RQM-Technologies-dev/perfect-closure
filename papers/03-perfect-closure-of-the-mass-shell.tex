\begin{abstract}
This paper introduces a mass-shell interpretation of Perfect Closure in a quaternionic formalization. Starting from $E^2-p^2c^2=m^2c^4$, we write momentum-energy in biquaternionic form and separate null propagation from massive closure by invariant norm.
\end{abstract}
\section{Thesis}
\textbf{Mass is the Perfect Closure of light.}
\section{Mass shell and quaternionic form}
\[
E^2-p^2c^2=m^2c^4,\qquad
\frac{E^2}{c^2}-|\mathbf p|^2=m^2c^2.
\]
Set
\[
P=\frac{E}{c}+I\mathbf p,\qquad I^2=-1,
\]
then
\[
P\bar P=\frac{E^2}{c^2}-|\mathbf p|^2,
\]
so
\[
P\bar P=m^2c^2.
\]
\section{Interpretation}
\begin{itemize}
\item $P\bar P=0$: null/open lightlike propagation.
\item $P\bar P>0$: nonzero invariant norm (massive closure).
\end{itemize}
\section{What this does and does not claim}
\textbf{Does not claim:} derivation of masses from first principles or a full QFT.\\
\textbf{Does claim:} the mass shell is a closure law in invariant norm form.
