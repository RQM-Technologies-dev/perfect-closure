\begin{abstract}
This paper advances the thesis that Perfect Closure is native to quaternionic geometry and that the classical critical line $\Re(s)=\tfrac12$ is its complex trace. We define quaternionic critical slices $q_u(t)=\tfrac12+ut$ across unit imaginary directions $u^2=-1$, distinguish affine slices from unit phase elements, and derive a six-step closure gate at real part $\tfrac12$ in phase form. We then formalize a slice-lifted completed zeta representation and a quaternionic closure residual. The proposal is interpretive and programmatic: it does not prove the Riemann Hypothesis and does not replace analytic number theory.
\end{abstract}

\section{Thesis}

\begin{quote}
Perfect Closure is visible in complex phase, but native to quaternionic orientation space. The Riemann critical line is one complex shadow of the quaternionic closure slice $\Re(q)=\tfrac12$.
\end{quote}

\section{Quaternionic and Complex Slices}

Let $u$ be a unit imaginary quaternion with $u^2=-1$.

\begin{definition}[Quaternionic critical slice]
\[
q_u(t)=\frac12+ut,\qquad t\in\mathbb R.
\]
\end{definition}

\begin{definition}[Complex slice associated to $u$]
\[
\mathbb C_u=\{a+ub:\;a,b\in\mathbb R,\;u^2=-1\}.
\]
\end{definition}

The usual critical line $s=\tfrac12+it$ is the special case $u=i$.

\begin{remark}
$\Re(s)=\tfrac12$ is not the whole closure geometry; it is the complex trace of the quaternionic hyperplane $\Re(q)=\tfrac12$.
\end{remark}

\section{Affine Slice vs. Unit Phase Form}

We distinguish:
\begin{itemize}
  \item Affine slice: $q=\tfrac12+ut$.
  \item Unit phase form: $Q=e^{u\phi}=\cos\phi+u\sin\phi$.
\end{itemize}

\section{Six-Step Closure Gate}

If $\Re(Q)=\tfrac12$, then $\cos\phi=\tfrac12$ and $\phi=\pi/3$. Hence
\[
Q=e^{u\pi/3},\qquad Q^3=e^{u\pi}=-1,\qquad Q^6=e^{u2\pi}=1.
\]
So $\Re(q)=\tfrac12$ acts as a closure gate in quaternionic phase geometry. This six-step law is already visible on the ordinary complex unit circle; quaternionic space makes it axis-free across imaginary directions.

\section{Slice-Lifted Completed Zeta and Residual}

Define
\[
\xi_H\!\left(\frac12+ut\right)
=
\Re\,\xi\!\left(\frac12+it\right)+u\Im\,\xi\!\left(\frac12+it\right),
\]
and
\[
C_{H,u}(t)=\left\|\xi_H\!\left(\frac12+ut\right)\right\|^2.
\]
A Perfect Closure event is $C_{H,u}(t)=0$.

\begin{remark}
The complex residual $|\xi(\tfrac12+it)|^2$ is one slice representation of this quaternionic residual.
\end{remark}
