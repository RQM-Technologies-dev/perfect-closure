\begin{abstract}
We develop the quaternionic-geometry layer of the Perfect Closure program. The critical-line condition in $\mathbb C$ is interpreted as a slice of the quaternionic condition $\Re(q)=\tfrac12$. We separate affine slices from unit-phase closure, derive the six-step gate $Q^6=1$ at $\Re(Q)=\tfrac12$, and define a slice-lifted completed-zeta residual that agrees with the complex residual on each slice. No proof of RH is claimed.
\end{abstract}

\section{Introduction and thesis}

\begin{quote}
Perfect Closure is visible in complex phase, but native to quaternionic orientation space.
\end{quote}

\section{Quaternions and complex slices}

A quaternion is $q=a+bi+cj+dk$ with $\Re(q)=a$. Let $u$ be unit imaginary ($u^2=-1$).

\begin{definition}[Complex slice]
\[
\mathbb C_u=\{a+ub:\;a,b\in\mathbb R\}.
\]
\end{definition}

The ordinary complex plane is the special slice $u=i$.

\section{Quaternionic critical slice}

\begin{definition}[Critical hyperslice]
\[
\Sigma_{1/2}=\{q\in\mathbb H:\Re(q)=1/2\}.
\]
\end{definition}

For fixed $u$, define
\[
q_u(t)=\frac12+ut,\qquad t\in\mathbb R.
\]
When $u=i$, this is $s=\tfrac12+it$. Hence the classical critical line is a complex trace of $\Sigma_{1/2}$.

\section{Affine slice versus unit phase}

Affine location and phase closure are distinct:
\[
q=\frac12+ut \quad\text{vs}\quad Q=e^{u\phi}=\cos\phi+u\sin\phi.
\]

\section{Six-step closure gate}

\begin{proposition}
If $\Re(Q)=\tfrac12$ for $Q=e^{u\phi}$, then $Q^3=-1$ and $Q^6=1$.
\end{proposition}

\begin{proof}
$\Re(Q)=\cos\phi=\tfrac12$ implies $\phi=\pi/3$ modulo $2\pi$. Then
\[
Q^3=e^{u\pi}=-1,\qquad Q^6=e^{u2\pi}=1.
\]
\end{proof}

\begin{remark}
The same six-step closure appears on the complex unit circle. Quaternionic space generalizes it from one axis to an axis-free family over $u\in S^2$.
\end{remark}

\section{Quaternionic prime phase}

\[
p^{-(1/2+ut)}=p^{-1/2}e^{-ut\log p},\qquad p^{-(1/2-ut)}=p^{-1/2}e^{+ut\log p}.
\]
This is the quaternionic square-root mirror.

\section{Slice-lifted completed zeta}

Define
\[
\xi_{\mathbb H}(1/2+ut)=\Re\,\xi(1/2+it)+u\Im\,\xi(1/2+it),
\]
and
\[
C_{\mathbb H,u}(t)=\|\xi_{\mathbb H}(1/2+ut)\|^2.
\]
Then $C_{\mathbb H,u}(t)=|\xi(1/2+it)|^2$ on each slice.
