\begin{abstract}
We develop the quaternionic-geometry layer of the Perfect Closure program. The complex critical-line condition is interpreted as a slice of a broader quaternionic condition $\Re(q)=\tfrac12$. We separate affine spectral location from unit-phase closure, derive the six-step gate $Q^6=1$ at $\Re(Q)=\tfrac12$, and define a slice-lifted residual agreeing with $|\Xi(t)|^2$ on each slice. No proof of RH is claimed.
\end{abstract}

\section{Quaternions and complex slices}
A quaternion is $q=a+bi+cj+dk$ with $\Re(q)=a$.

\begin{definition}[Complex slice]
For a unit imaginary quaternion $\mathbf u$ with $\mathbf u^2=-1$,
\[
\mathbb C_{\mathbf u}=\{a+\mathbf u b:\;a,b\in\mathbb R\}.
\]
\end{definition}

This gives a family of complex planes indexed by axis $\mathbf u$.

\section{Quaternionic critical slice}
\begin{definition}[Critical hyperslice]
\[
\Sigma_{1/2}=\{q\in\mathbb H:\Re(q)=1/2\}.
\]
\end{definition}

For fixed $\mathbf u$,
\[
q_{\mathbf u}(t)=\frac12+\mathbf u t.
\]
When $\mathbf u=i$, this becomes the classical critical line.

\section{Affine slice versus unit phase}
Keep distinct:
\[
q=\frac12+\mathbf u t \quad\text{(affine location)},\qquad
Q=e^{\mathbf u\phi}=\cos\phi+\mathbf u\sin\phi \quad\text{(unit phase)}.
\]
The distinction prevents geometric and multiplicative claims from being conflated.

\section{Six-step closure gate}
\begin{proposition}
If $\Re(Q)=\tfrac12$ for $Q=e^{\mathbf u\phi}$, then $Q^3=-1$ and $Q^6=1$.
\end{proposition}

\begin{proof}
$\Re(Q)=\cos\phi=\tfrac12$ implies $\phi=\pi/3$ modulo $2\pi$, hence
\[
Q^3=e^{\mathbf u\pi}=-1,\qquad Q^6=e^{\mathbf u2\pi}=1.
\]
\end{proof}

\begin{remark}
This is a geometric closure cadence. It does not by itself generate or prove zeta-zero placement.
\end{remark}

\section{Quaternionic prime phase}
\[
p^{-(1/2+\mathbf u t)}=p^{-1/2}e^{-\mathbf u t\log p},\qquad p^{-(1/2-\mathbf u t)}=p^{-1/2}e^{+\mathbf u t\log p}.
\]
This is the quaternionic form of the square-root mirror from Paper 1.

\section{Slice-lifted completed zeta}
\[
\Xi(t)=\xi\!\left(\frac12+it\right),
\]
\[
\xi_{\mathbb H}(1/2+\mathbf u t)=\Re\,\Xi(t)+\mathbf u\Im\,\Xi(t),
\]
\[
C_{\mathbb H,\mathbf u}(t)=\|\xi_{\mathbb H}(1/2+\mathbf u t)\|^2=|\Xi(t)|^2.
\]
This shows slice compatibility and prepares operator notation.
