\begin{abstract}
This paper gives the geometry layer of the series. The complex variable $\tfrac12+it$ is lifted to quaternionic slices $q_{\mathbf u}(t)=\tfrac12+\mathbf u t$ with $\mathbf u^2=-1$, where the classical line is the slice $\mathbf u=i$. We separate affine slice location from unit-phase motion and record the six-step cadence $Q^3=-1$, $Q^6=1$ at $\Re(Q)=\tfrac12$ as geometric intuition only. No proof of RH is claimed.
\end{abstract}

\section*{Reader Guide}
The role of Paper 2 is not to prove zeta facts, but to show that the complex spectral trace $\tfrac12+it$ can be lifted to the quaternionic family $\tfrac12+\mathbf u t$.

\Needspace{10\baselineskip}
\section{Quaternionic slices and critical hyperslice}
\begin{definition}[Complex slice]
For a unit imaginary quaternion $\mathbf u$ with $\mathbf u^2=-1$,
\[
\mathbb C_{\mathbf u}=\{a+\mathbf u b:\;a,b\in\mathbb R\}.
\]
\end{definition}

Define the critical hyperslice
\[
\Sigma_{1/2}=\{q\in\mathbb H:\Re(q)=1/2\},
\]
and affine family
\[
q_{\mathbf u}(t)=\frac12+\mathbf u t.
\]
For $\mathbf u=i$, this is the classical critical line.

\section{Affine slice versus unit phase}
Keep distinct
\[
q=\frac12+\mathbf u t \quad(\text{affine slice}),\qquad
Q=e^{\mathbf u\phi}=\cos\phi+\mathbf u\sin\phi \quad(\text{unit phase}).
\]

\Needspace{10\baselineskip}
\section{Six-step cadence as geometric intuition}
If $\Re(Q)=\tfrac12$, then $\cos\phi=\tfrac12$, hence $\phi=\pi/3$ modulo $2\pi$, so
\[
Q^3=-1,\qquad Q^6=1.
\]
\begin{remark}
This cadence helps visualize closure at real part $\tfrac12$, but the bridge to mass shells is carried by the eigenvalue operator in Paper 4.
\end{remark}

\section{Quaternionic lift of prime phase and residual}
\[
p^{-(1/2+\mathbf u t)}=p^{-1/2}e^{-\mathbf u t\log p}.
\]
With
\[
\Xi(t)=\xi\!\left(\frac12+it\right),
\]
a compatible slice lift is
\[
\xi_{\mathbb H}(1/2+\mathbf u t)=\Re\Xi(t)+\mathbf u\Im\Xi(t),
\]
with
\[
\|\xi_{\mathbb H}(1/2+\mathbf u t)\|^2=|\Xi(t)|^2.
\]
