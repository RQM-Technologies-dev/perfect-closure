\begin{abstract}
We present the operator bridge of the Perfect Closure series. A quaternionic dilation generator $A_u=u(\rho\tfrac{d}{d\rho}+\tfrac12)$ has eigenmodes $\chi_t(\rho)=\rho^{-1/2-ut}$. Defining $Z_u=\Xi(A_u)$ with $\Xi(t)=\xi(\tfrac12+it)$, each critical-line ordinate $t_n$ gives a kernel mode $Z_u\chi_{t_n}=0$. A scale map then yields $m_n=\hbar|t_n|/(cL_*)$. The construction is formal and does not prove RH or derive Standard Model masses.
\end{abstract}

\section{Internal scale space}
Let $\rho>0$ and $y=\log\rho$. Prime phases naturally involve $\log p$.

\section{Quaternionic dilation generator}
Define
\[
A_u=u\left(\rho\frac{d}{d\rho}+\frac12\right),\qquad u^2=-1.
\]
Set
\[
\chi_t(\rho)=\rho^{-1/2-ut}=\rho^{-1/2}e^{-ut\log\rho}.
\]
Then
\[
\left(\rho\frac{d}{d\rho}+\frac12\right)\chi_t=-ut\chi_t,
\qquad
A_u\chi_t=t\chi_t.
\]

\section{Prime evaluation}
At $\rho=p$,
\[
\chi_t(p)=p^{-1/2-ut}=p^{-1/2}e^{-ut\log p}.
\]
For $u=i$, this recovers the complex square-root prime phase.

\section{Zeta closure selector}
Define
\[
\Xi(t)=\xi\!\left(\frac12+it\right),\qquad Z_u=\Xi(A_u)
\]
(formal functional calculus on a suitable spectral domain).
If $\Xi(t_n)=0$, then
\[
Z_u\chi_{t_n}=\Xi(A_u)\chi_{t_n}=\Xi(t_n)\chi_{t_n}=0.
\]

\section{Closure period in log-scale}
Mode phase is $e^{-ut_n\log\rho}$. Full closure and six-step increment are
\[
\Lambda_n=\frac{2\pi}{t_n},\qquad \Delta_n=\frac{\pi}{3t_n}.
\]

\section{Mass-shell lift}
Let
\[

Matching $k_C=mc/\hbar$ gives
\[
m_n=\frac{\hbar|t_n|}{cL_*}.
\]
With $P=E/c+I\mathbf p$,
\[
P\bar P=\left(\frac{\hbar t_n}{L_*}\right)^2.

\section{Main theorem}

\begin{theorem}[Closure eigenvalue lift]
If $A_u\chi_{t_n}=t_n\chi_{t_n}$ and $\Xi(t_n)=0$, then $\chi_{t_n}\in\ker\Xi(A_u)$. Under scale map $L_*$, the same eigenvalue defines
\[
m_n=\frac{\hbar|t_n|}{cL_*}.
\]
\end{theorem}

\begin{proof}
First,
\[
\Xi(A_u)\chi_{t_n}=\Xi(t_n)\chi_{t_n}=0.
\]
Second, define $k_n=t_n/L_*$. Identifying $k_n$ with Compton curvature $k_C=mc/\hbar$ gives $m_n=\hbar|t_n|/(cL_*)$.
\end{proof}
